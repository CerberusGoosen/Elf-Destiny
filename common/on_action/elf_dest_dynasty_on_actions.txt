on_dynasty_created = {
	on_actions = {
		assign_elf_coa
	}
}

assign_elf_coa = {
	effect = {
		debug_log = "assign_elf_coa"
		debug_log_scopes = yes

		if = {
			limit = {
				dynast = {
					is_aeluran_sister_or_higher = yes
				}
			}
			debug_log = "assign_elf_coa AELURAN"

			generate_coa = aeluran_title
		}
		# scope:dynasty = { generate_coa = aeluran_title }
	}
}
